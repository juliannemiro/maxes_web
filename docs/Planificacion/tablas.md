CREATE TABLE rubro (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL,
    nombre VARCHAR(100) NULL,
    activo BOOLEAN DEFAULT TRUE
);

CREATE TABLE pedido (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fecha TIMESTAMPTZ NULL DEFAULT CURRENT_TIMESTAMP,
    cliente_nro VARCHAR(50) NULL,
    nombre VARCHAR(100) NULL,
    apellido VARCHAR(100) NULL,
    doc_tipo VARCHAR(20) NULL,
    doc_numero VARCHAR(20) NULL,
    cuit VARCHAR(20) NULL,
    email_pedido VARCHAR(150) NULL,
    celular_pedido VARCHAR(50) NULL,
    cant_productos INT NULL,
    cant_unidades INT NULL,
    tipo_compra VARCHAR(20) NULL,
    monto_total NUMERIC(18,2) NULL,
    tipo_despacho VARCHAR(20) NULL,
    localidad VARCHAR(150) NULL,
    observaciones TEXT NULL,
    estado VARCHAR(30) NULL DEFAULT 'nuevo'
);

CREATE TABLE articulo_web (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    articulo_id_origen INT NOT NULL,
    articulo_cod VARCHAR(20) NOT NULL UNIQUE,
    articulo_des VARCHAR(30) NULL,
    articulo_texto_web TEXT NULL,
    articulo_imagen_id INT NULL,
    precio_mayorista NUMERIC(18,2) NULL,
    precio_minorista NUMERIC(18,2) NULL,
    rubro_id INT REFERENCES rubro(id) ON DELETE SET NULL,
    proveedor_des VARCHAR(20) NULL,
    stock_web INT NULL,
    destacado CHAR(1) NOT NULL DEFAULT 'N',
    visible CHAR(1) NOT NULL DEFAULT 'S',
    fecha_publicacion TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE pedido_detalle (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    pedido_id INT NOT NULL REFERENCES pedido(id) ON DELETE CASCADE,
    articulo_id INT REFERENCES articulo(id) ON DELETE SET NULL,
    cantidad INT NOT NULL,
    precio_unitario NUMERIC(18,2) NOT NULL,
    articulo_comentario VARCHAR(20) NULL
);

CREATE TABLE articulo_imagen_web (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    articulo_id INT NOT NULL REFERENCES articulo(id) ON DELETE CASCADE,
    imagen_url VARCHAR(500) NOT NULL,
    orden INT DEFAULT 0
);

CREATE TABLE carrusel_home (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    titulo VARCHAR(150) NULL,
    link_destino VARCHAR(250) NULL,
    imagen_url VARCHAR(500) NOT NULL,
    activo BOOLEAN DEFAULT TRUE,
    orden INT DEFAULT 0
);

CREATE TABLE configuracion (
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    whatsapp_contacto VARCHAR(50) NOT NULL,
    direccion_local VARCHAR(250) NOT NULL,
    email_notificaciones VARCHAR(150) NULL,
    mantenimiento BOOLEAN DEFAULT FALSE
);
